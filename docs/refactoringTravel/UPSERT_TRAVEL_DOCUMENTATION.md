interface TipConfig {
  condition: (context: any) => boolean;
  icon: string;
  text: string;
  tone: 'info' | 'success' | 'warning';
}

export function ContextualTips({ context, tips }: { context: any; tips: TipConfig[] }) {
  const activeTip = useMemo(() => {
    return tips.find((tip) => tip.condition(context));
  }, [context, tips]);
  
  if (!activeTip) return null;
  
  return (
    <View style={[styles.tipBanner, styles[`tipBanner${activeTip.tone}`]]}>
      <Icon name={activeTip.icon} color={colors[activeTip.tone]} />
      <Text>{activeTip.text}</Text>
    </View>
  );
}
