function escapeHtml(s: string) {
    return (s ?? '')
        .replace(/&/g, '&amp;').replace(/</g, '&lt;')
        .replace(/>/g, '&gt;').replace(/"/g, '&quot;')
}
 
async function makeQR(url: string) {
    try {

}
